main.c — ELIO Motor BLE Control 
// ELIO Board — Nordic nRF52 SDK example
// Flash via neoFlash SWD in < 3 seconds

#include "nrf_gpio.h"
#include "ble_nus.h"
#include "elio_motor.h"

// ELIO port definitions
#define ELIO_DC1_PWM    NRF_GPIO_PIN_MAP(0, 26)
#define ELIO_DC2_PWM    NRF_GPIO_PIN_MAP(0, 27)
#define ELIO_BLE_NAME   "ELIO_DEVICE"

void ble_data_handler(uint8_t *data, uint16_t len) {
    // Parse ELIO 15-byte command packet
    if (data[0] == STX && data[len-1] == ETX) {
        elio_motor_set(data[3], data[4]);  // DC1, DC2
        elio_servo_set(data[5], data[6]);  // SV1, SV2
    }
}
