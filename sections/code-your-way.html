nRF52840 BLE Dongle

When coding from a PC without built-in Bluetooth, plug in the nRF52840 USB Dongle — it acts as a BLE host so Scratch, Entry, and Python can communicate with ELIO wirelessly.

from elio import Elio

board = Elio()
board.connect()           # Bluetooth connect
board.dc(1, speed=80)     # DC Motor 1 → 80% forward
board.servo(1, angle=45)  # Servo 1 → 45 degrees
board.io(1, value=100)    # IO Port 1 → LED full bright
#include <Elio.h>

Elio elio;

void setup() {
  elio.begin();
}

void loop() {
  elio.dc(1, 80);      // DC Motor 1: 80% speed
  elio.servo(1, 45);   // Servo 1: 45 degrees
  delay(1000);
  elio.stop();
}
